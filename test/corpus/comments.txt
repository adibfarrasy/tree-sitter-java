javadoc comments
====================================

/**
 * This is a javadoc comment
 */
class Foo {}

/** Single line javadoc */
interface Bar {}

/**
 * Method documentation
 * @param x the parameter
 * @return the result
 */
int method(int x) { return x; }

class Foo {
  /** Constructor docs */
  Foo() {}
}

/**
 * Enum documentation
 */
enum Color { RED, GREEN }

class Foo {
    /** Field docs */
    String name;
}
